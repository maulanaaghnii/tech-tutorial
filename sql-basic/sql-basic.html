SQL Dasar

Pendahuluan

Structured Query Language, atau SQL, adalah bahasa standar yang digunakan untuk berinteraksi dengan database relasional. Baik Anda mengambil data, memasukkan record baru, memperbarui informasi yang ada, atau menghapus entri yang tidak diperlukan, SQL adalah alat penting yang membuat semuanya mungkin.

Dalam modul ini, kita akan mempelajari konsep dasar SQLâ€”cocok untuk pemula. Anda tidak memerlukan pengalaman sebelumnya dengan database; cukup bawa rasa ingin tahu dan kemauan untuk belajar. Di akhir kursus ini, Anda akan memahami cara menulis query sederhana, memfilter hasil, mengurutkan data, dan melakukan operasi dasar yang menjadi fondasi untuk bekerja dengan sistem database apa pun.

Mari kita mulai perjalanan Anda untuk menguasai SQL!


Membuat database baru


CREATE DATABASE nama_database;

Menggunakan database


USE nama_database;

Membuat tabel baru


CREATE TABLE products (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    price DECIMAL(10,2),
    category VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Menambah satu record


INSERT INTO products (name, price, category) 
VALUES ('Laptop', 15000000, 'Elektronik');

Menambah beberapa record


INSERT INTO products (name, price, category) VALUES 
    ('Mouse', 200000, 'Elektronik'),
    ('Keyboard', 500000, 'Elektronik');

Membaca data dengan kondisi yang berbeda


SELECT * FROM products;  -- Semua kolom
SELECT name, price FROM products;  -- Kolom tertentu
SELECT DISTINCT category FROM products;  -- Nilai unik

Mengupdate data


UPDATE products 
SET price = price * 1.1  -- Meningkatkan harga sebesar 10%
WHERE category = 'Elektronik';

Menghapus data


DELETE FROM products WHERE price < 100000;

Menggunakan operator perbandingan


SELECT * FROM products 
WHERE price >= 1000000 & price <= 5000000;

BETWEEN untuk rentang


SELECT * FROM products 
WHERE price BETWEEN 1000000 & 5000000;

IN untuk beberapa nilai


SELECT * FROM products 
WHERE category IN ('Elektronik', 'Furniture');

LIKE untuk pencarian pola


SELECT * FROM products 
WHERE name LIKE '%phone%';  -- Mencari nama dengan 'phone'
WHERE name LIKE '_ouse';    -- Mencari kata yang diakhiri dengan 'ouse'

Mengelompokkan hasil berdasarkan kategori


SELECT 
    category,
    COUNT(*) as jumlah_produk,
    AVG(price) as harga_rata_rata
FROM products
GROUP BY category
HAVING COUNT(*) > 5;  -- Filter setelah mengelompokkan

Inner Join


SELECT 
    p.name,
    c.category_name,
    p.price
FROM products p
INNER JOIN categories c ON p.category_id = c.id;

Left Join


SELECT 
    p.name,
    c.category_name,
    p.price
FROM products p
LEFT JOIN categories c ON p.category_id = c.id;

Right Join


SELECT 
    p.name,
    c.category_name,
    p.price
FROM products p
RIGHT JOIN categories c ON p.category_id = c.id;

Mengelompokkan dengan kondisi


SELECT 
    category,
    COUNT(*) as jumlah_produk,
    AVG(price) as harga_rata_rata
FROM products
GROUP BY category
HAVING COUNT(*) > 5;  -- Filter setelah mengelompokkan

Subquery dalam SELECT


SELECT name, price,
    (SELECT AVG(price) FROM products) as harga_rata_rata
FROM products;

Subquery dengan WHERE


SELECT * FROM products 
WHERE price > (
    SELECT AVG(price) FROM products
);

Subquery dalam FROM


SELECT category, harga_rata_rata
FROM (
    SELECT category, AVG(price) as harga_rata_rata
    FROM products
    GROUP BY category
) as category_stats
WHERE harga_rata_rata > 1000000;

Membuat indeks untuk optimasi query


CREATE INDEX idx_category ON products(category);
CREATE INDEX idx_price ON products(price);

Indeks komposit untuk beberapa kolom


CREATE INDEX idx_cat_price ON products(category, price);

Menghapus indeks


DROP INDEX idx_category ON products;

Membuat view


CREATE VIEW produk_mahal AS
SELECT * FROM products
WHERE price > 5000000;

Membuat prosedur tersimpan


DELIMITER //
CREATE PROCEDURE update_harga(
    IN nama_kategori VARCHAR(50),
    IN persentase_kenaikan INT
)
BEGIN
    UPDATE products
    SET price = price * (1 + persentase_kenaikan / 100)
    WHERE category = nama_kategori;
END//
